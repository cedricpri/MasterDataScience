# use a for loop to generate a walk
position = [0, 0] #Initial position x, y
walk = [] #List in which we will keep the walk obtained
steps = 10000

for i in range(steps):
    random1 = random.randint(0,3) #Gives the direction of the movement (0 is to the left, 1 to the top, 2 to the right, 3 to the bottom)

    if random1 == 0:
        movement = [-1, 0]
    elif random1 == 1:
        movement = [0, 1]
    elif random1 == 2:
        movement = [1, 0]
    else:
        movement = [0, -1]
    
    newPosition = [position[0] + movement[0], position[1] + movement[1]]
    walk.append(newPosition)
    position = newPosition
